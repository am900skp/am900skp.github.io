    ahead-of-time语言

    安装验证 rustc –version

    rust帮助文档: rustup doc

    编译:

    • rustc main.rs

    • 在windows上还会生成一个.pdb文件

    • rustc只适合编译简单的rust程序

      • Cargo :

        • cargo new hello_cargo
        • 可以使用其他的VCS,或不适用VCS,cargo new 的时候使用 -vcs这个flag
        • 代码的包称作crate
        • cargo.toml

      • Cargo构建项目

        • cargo build
          • cargo.lock,不要乱动这个文件

      • 构建和运行Cargo项目

        • cargo run

      • cargo check

        • 检查代码,确保能通过编译,但是不产生任何可执行文件

      • 为发布构建

        • cargo build –release

      • 尽量用Cargo

    [package]
name = "hello_cargo"
version = "0.1.0"
edition = "2021"

# See more keys and their definitions at https://doc.rust-lang.org/cargo/reference/manifest.html

[dependencies]
rand = "0.8.4"

    use std::io;    //prelude
use std::cmp::Ordering;
use rand::Rng;  //trait

fn main() {
    
    println!("guess the number of bytes to read_line");
    let secret_number = rand::thread_rng().gen_range(1..1024);
    println!("secret number: {}", secret_number);
    let _res = 1;          //mut可以改变
    let mut guess = String::new();  //返回字符串新的实例
    //new:创建一个空的字符串
    //& : 表示参数是引用类型
    io::stdin().read_line(&mut guess).expect("Unable to read_line");
    //io:Result Ok, Err 
    //{}: 占位符, 对应后面变量的值
    let guess: u32 = guess.trim().parse().expect("Please type a number!");
    println!("number: {}",guess);
    match guess.cmp(&secret_number) {
        Ordering::Less => println!("Too small!"),
        Ordering::Greater => println!("To big"),
        Ordering::Equal => println!("To Win!"),
    }
}

    use rand::Rng;
use std::cmp::Ordering;
use std::io; //prelude //trait

fn main() {
    let secret_number = rand::thread_rng().gen_range(1..1024);
    loop {
        println!("guess the number of bytes to read_line");
        let mut guess = String::new(); //返回字符串新的实例
        io::stdin().read_line(&mut guess).expect("Unable to read_line");
        let guess: u32 = match guess.trim().parse() {
            Ok(num) => num,
            Err(_) => continue,
        };
        println!("number: {}", guess);
        match guess.cmp(&secret_number) {
            Ordering::Less => println!("Too small!"),
            Ordering::Greater => println!("To big"),
            Ordering::Equal => {
                println!("To Win!");
                break;
            }
        }
    }
}
