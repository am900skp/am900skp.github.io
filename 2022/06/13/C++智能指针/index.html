    C++智能指针

    智能指针本质是一个类的对象，使用的时候不会创建这个类指针，只会使用这个类的对象

    智能指针原理是通过对象的死亡会自动调用析构函数这个功能去实现

    • 构造函数: 管理new的对象
    • 析构函数: 析构new的对象

    C++共享型智能指针

    • 允许拷贝构造
    • 允许多个智能指针对象管理同一段内存
    // shared_ptr
#include <iostream>
#include <memory>
#include <string>
using namespace std;
class MM {
   public:
    MM(string name, int age) : name(name), age(age) {}
    void print() { 
        cout << name << " " << age << endl; 
    }
    ~MM() {
        cout << "析构过程" << endl; 
    }

   protected:
    string name;
    int age;
};
class Boy {
   public:
    Boy() { 
        cout << "Boy" << endl;
    }
    ~Boy() {
        cout << "~Boy" << endl;
    }
};

void printData(shared_ptr<MM> mm) {
    mm->print();
}
auto returnPtr() {
    shared_ptr<MM> mm(new MM("小芳", 18));
    return mm;
}
void FreeFile(FILE*& fp) {
    cout << "关闭文件成功" << endl;
    fclose(fp);
}

void test_shared_ptr() {
    // 1.管理基本数据
    shared_ptr<int> intData(new int(100));
    if (!intData) {
        cout << "初始化之智能指针失败!" << endl;
        return;
    }
    cout << *intData << endl;
    //获取管理对象地址
    // int* x = intData.get();
    // cout << *x << endl;
    cout << "size:" << intData.use_count() << endl;
    shared_ptr<int> p2(intData);
    cout << "size:" << intData.use_count() << endl;
    // swap
    // reset
    // make_shared函数的参数由构造函数决定
    // 2.管理自定义类型
    shared_ptr<string> str = make_shared<string>("ILoveyou");
    cout << *str << endl;
    {
        shared_ptr<MM> mm(new MM("张三", 19));
        mm->print();
        mm.reset(new MM("小芳", 18));
        mm->print();
        shared_ptr<MM> mm2 = make_shared<MM>("小芳", 20);
        mm2->print();
    }
    // 3.智能指针常用场景
    shared_ptr<MM> temp(new MM("亲爱的", 28));
    printData(temp);
    printData(returnPtr());
    // 4.特殊释放形式的内存，需要自己手动写删除器
    // C语言文件指针
    { shared_ptr<FILE> pf(fopen("1.txt", "w+"), FreeFile); }

    {
        shared_ptr<Boy> p(new Boy[10], [](Boy*& p) { delete[] p; });
    }
    {
        shared_ptr<int> pint(new int(999));
        int* pp = pint.get();
        *pp = 666;
        cout << *pint << endl;
        // delete pp;	//不能直接释放，会导致二次析构问题
        // cout << *pint << endl;
    }
}
int main(int argc, char** argv) {
    test_shared_ptr();

    return 0;
}

    C++弱引用型智能指针

    • weak_ptr 是弱引用指针，不会累计计数
    • weak_ptr不能直接管理对象，只能shared_ptr或者weak_ptr构造而来
    • 目的是为了解决shared_ptr循环引用的时候内存未释放问题
    • 不可以使用* 或者->访问对象，只能通过lock函数访问shared_ptr对象，再去访问数据
    // weak_ptr
#include <iostream>
#include <memory>
#include <string>
using namespace std;
class B;
class A {
   public:
    A() { cout << "A" << endl; }
    ~A() { cout << "~A" << endl; }
    // shared_ptr<B> b;
    weak_ptr<B> b;
};
class B {
   public:
    B() { cout << "B" << endl; }
    ~B() { cout << "~B" << endl; }
    // shared_ptr<A> a;
    weak_ptr<A> a;
};

void test_weak_ptr() {
    {
        shared_ptr<A> ao = make_shared<A>();
        shared_ptr<B> bo = make_shared<B>();
        cout << ao.use_count() << endl;
        cout << bo.use_count() << endl;
        ao->b = bo;
        bo->a = ao;
        cout << ao.use_count() << endl;
        cout << bo.use_count() << endl;
    }

    {
        shared_ptr<int> s(new int(999));
        cout << "count:" << s.use_count() << endl;
        weak_ptr<int> w(s);
        cout << "count:" << w.use_count() << endl;
        cout << "count:" << s.use_count() << endl;
        cout << *w.lock() << endl;
    }
}
int main(int argc, char** argv) {
    // test_shared_ptr();
    test_weak_ptr();

    return 0;
}

    C++独享型智能指针

    • 禁止拷贝和赋值(通过删除拷贝构造函数和赋值重载)
    • 可以通过转交所有权的方式让一个指针去管理对象
      • move函数移交所有权限
      • 内置reset函数结合release函数去移交所有权限
    // unique_ptr
#include <iostream>
#include <memory>
#include <string>
using namespace std;
class MM {
   public:
    MM(string name, int age) : name(name), age(age) {}
    void print() {
        cout << name << " " << age << endl;
    }
    ~MM() { 
        cout << "析构过程" << endl;
    }

   protected:
    string name;
    int age;
};
class Boy {
   public:
    Boy() { 
        cout << "Boy" << endl;
    }
    ~Boy() { 
        cout << "~Boy" << endl; 
    }
};
void test_unique_ptr() {
    unique_ptr<int> ptr(new int(999));
    cout << *ptr << endl;
    // unique_ptr<int> ptr2(ptr);
    //转交所有权
    unique_ptr<int> ptr2;
    ptr2 = move(ptr);
    // cout << *ptr << endl;		//没有值了
    cout << *ptr2 << endl;
    unique_ptr<int> ptr3;
    ptr3.reset(ptr2.release());
    cout << *ptr3 << endl;
    auto p = ptr3.get();
    // delete p;		//导致二次析构问题
    {
        unique_ptr<MM> pMM(new MM("小美", 19));
        pMM->print();
        // unique_ptr<管理对象类型,删除器函数指针类型>
        using FUNC = void (*)(Boy*&);
        unique_ptr<Boy, FUNC> pBoy(new Boy[3],
                                   [](Boy*& pBoy) { delete[] pBoy; });
        //不能用auto
        // unique_ptr<Boy, auto> pBoy2(new Boy[3], [](Boy*& pBoy) {delete[]
        // pBoy; });
    }
}
int main(int argc, char** argv) {
    // test_shared_ptr();
    // test_weak_ptr();
    test_unique_ptr();
    return 0;
}
